### 3.2 ì„ ì • SQL

```sql
WITH ranked_files AS (
    SELECT
        id,
        catalog_title,
        episode_title,
        version_type,
        ROW_NUMBER() OVER (
            PARTITION BY catalog_title, episode_title
            ORDER BY
                CASE version_type
                    WHEN 'stream' THEN 1
                    WHEN 'clean' THEN 2
                    WHEN 'final_edit' THEN 3
                    WHEN 'mastered' THEN 4
                    WHEN 'nobug' THEN 5
                    WHEN 'generic' THEN 6
                    WHEN 'pgm' THEN 7
                    WHEN 'hires' THEN 8
                    ELSE 9
                END,
                file_size_bytes DESC NULLS LAST
        ) as rn
    FROM pokervod.video_files
    WHERE is_hidden = false
)
UPDATE pokervod.video_files
SET is_catalog_item = (id IN (SELECT id FROM ranked_files WHERE rn = 1));
```
